run.sh
#!/bin/bash
cd /home/youruser/myapp
source venv/bin/activate
export FLASK_APP=app.py
export FLASK_ENV=production
flask run --host=0.0.0.0 --port=8000


Make it executable:
chmod +x /home/youruser/myapp/run.sh

Create a systemd Service File

Create a new file:
sudo nano /etc/systemd/system/flaskapp.service

Paste this in:

[Unit]
Description=Flask App Service
After=network.target

[Service]
User=youruser
WorkingDirectory=/home/youruser/myapp
ExecStart=/home/youruser/myapp/run.sh
Restart=always
Environment=PATH=/home/youruser/myapp/venv/bin
Environment=FLASK_ENV=production

[Install]
WantedBy=multi-user.target


Reload systemd and Enable Service
sudo systemctl daemon-reexec
sudo systemctl daemon-reload
sudo systemctl enable flaskapp.service
sudo systemctl start flaskapp.service

sudo systemctl status flaskapp.service
journalctl -u flaskapp.service -e
